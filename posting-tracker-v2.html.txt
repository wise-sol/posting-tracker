<script>
const STORAGE_KEY = "posting_tracks_v2";
const TRACK_COLORS = ["#00e5ff","#ff6b35","#7fff00","#ff3f8e","#ffd600","#b388ff","#00ffcc"];
const AREA_COLORS  = ["#ff9100","#e040fb","#40c4ff","#69f0ae","#ffea00","#ff5252"];
let map, currentPolyline, currentMarker;
let recording = false, watchId = null, timerId = null;
let startTime = 0, elapsed = 0, currentPoints = [];
let tracks = [], pendingTrack = null, areaLayers = [];
let trackPolylines = {}, selectedTrackId = null;

window.addEventListener("load", () => { initMap(); loadTracks(); renderTrackList(); });

function initMap() {
  map = L.map("map", { center: [35.011636, 135.768029], zoom: 14, zoomControl: true });
  L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
    attribution: '© <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>', maxZoom: 19,
  }).addTo(map);
}

function switchTab(id) {
  document.querySelectorAll(".tab-btn").forEach((b, i) => { b.classList.toggle("active", ["rec","tracks","area"][i] === id); });
  document.querySelectorAll(".tab-content").forEach(el => el.classList.remove("active"));
  document.getElementById(`tab-${id}`).classList.add("active");
}

function startRecording() {
  if (!navigator.geolocation) { toast("GPSが使えません"); return; }
  recording = true; currentPoints = []; startTime = Date.now(); elapsed = 0;
  document.getElementById("start-btn").style.display = "none";
  document.getElementById("stop-btn").style.display = "block";
  document.getElementById("rec-badge").classList.add("visible");
  document.getElementById("stats-overlay").classList.add("visible");
  const color = TRACK_COLORS[tracks.length % TRACK_COLORS.length];
  currentPolyline = L.polyline([], { color, weight: 4, opacity: 0.9, lineCap: "round", lineJoin: "round" }).addTo(map);
  const icon = L.divIcon({ className: "", html: `<div style="width:14px;height:14px;background:${color};border:2px solid #fff;border-radius:50%;box-shadow:0 0 8px ${color}"></div>`, iconSize: [14,14], iconAnchor: [7,7] });
  currentMarker = L.marker([35.011636, 135.768029], { icon }).addTo(map);
  watchId = navigator.geolocation.watchPosition(onPosition, err => toast("GPS: " + err.message), { enableHighAccuracy: true, maximumAge: 3000, timeout: 15000 });
  timerId = setInterval(() => { elapsed = Math.floor((Date.now() - startTime) / 1000); document.getElementById("stat-time").textContent = formatTime(elapsed); }, 1000);
  toast("記録開始しました");
}

function onPosition(pos) {
  const lat = pos.coords.latitude, lon = pos.coords.longitude, acc = Math.round(pos.coords.accuracy);
  currentPoints.push({ lat, lon, ts: pos.timestamp });
  currentPolyline.addLatLng([lat, lon]);
  currentMarker.setLatLng([lat, lon]);
  map.panTo([lat, lon]);
  const dist = calcDistance(currentPoints);
  document.getElementById("stat-dist").textContent = (dist / 1000).toFixed(2);
  document.getElementById("stat-pts").textContent = currentPoints.length;
  document.getElementById("accuracy-line").textContent = `GPS: ±${acc}m`;
}

function stopRecording() {
  if (!recording) return;
  navigator.geolocation.clearWatch(watchId);
  clearInterval(timerId);
  recording = false;
  document.getElementById("start-btn").style.display = "block";
  document.getElementById("stop-btn").style.display = "none";
  document.getElementById("rec-badge").classList.remove("visible");
  if (currentPoints.length < 2) {
    toast("記録が短すぎます");
    if (currentPolyline) map.removeLayer(currentPolyline);
    if (currentMarker) map.removeLayer(currentMarker);
    document.getElementById("stats-overlay").classList.remove("visible");
    return;
  }
  const dist = calcDistance(currentPoints);
  const color = TRACK_COLORS[tracks.length % TRACK_COLORS.length];
  pendingTrack = { id: Date.now(), points: currentPoints, duration: elapsed, distance: dist, color, date: new Date().toISOString(), name: `配布 ${new Date().toLocaleDateString("ja-JP")}` };
  document.getElementById("modal-meta").textContent = `${(dist/1000).toFixed(2)} km ／ ${formatTime(elapsed)}`;
  document.getElementById("modal-input").value = pendingTrack.name;
  document.getElementById("modal-overlay").classList.add("visible");
  document.getElementById("modal-input").focus();
}

function saveTrack() {
  if (!pendingTrack) return;
  pendingTrack.name = document.getElementById("modal-input").value || pendingTrack.name;
  tracks.unshift(pendingTrack); saveTracks();
  trackPolylines[pendingTrack.id] = currentPolyline;
  if (currentMarker) map.removeLayer(currentMarker);
  pendingTrack = null;
  document.getElementById("modal-overlay").classList.remove("visible");
  document.getElementById("stats-overlay").classList.remove("visible");
  renderTrackList(); switchTab("tracks"); toast("保存しました");
}

function discardTrack() {
  if (currentPolyline) map.removeLayer(currentPolyline);
  if (currentMarker) map.removeLayer(currentMarker);
  pendingTrack = null;
  document.getElementById("modal-overlay").classList.remove("visible");
  document.getElementById("stats-overlay").classList.remove("visible");
}

function loadTracks() {
  try { tracks = JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; } catch { tracks = []; }
  tracks.forEach(t => {
    if (t.points && t.points.length >= 2) {
      const pl = L.polyline(t.points.map(p => [p.lat, p.lon]), { color: t.color, weight: 3, opacity: 0.7 }).addTo(map);
      trackPolylines[t.id] = pl;
    }
  });
}

function saveTracks() { try { localStorage.setItem(STORAGE_KEY, JSON.stringify(tracks)); } catch { toast("保存容量が不足しています"); } }

function renderTrackList() {
  const el = document.getElementById("track-list");
  if (tracks.length === 0) { el.innerHTML = `<div style="color:var(--muted);font-size:12px;text-align:center;padding:20px">記録がありません</div>`; return; }
  el.innerHTML = tracks.map(t => `
    <div class="track-item ${selectedTrackId === t.id ? 'selected' : ''}" onclick="focusTrack(${t.id})">
      <div class="track-color-dot" style="background:${t.color}"></div>
      <div class="track-info"><div class="track-name">${t.name}</div><div class="track-meta">${new Date(t.date).toLocaleDateString("ja-JP")} · ${(t.distance/1000).toFixed(2)}km · ${formatTime(t.duration)}</div></div>
      <div class="track-actions">
        <button class="btn-sm btn-sm-cyan" onclick="event.stopPropagation();exportTrack(${t.id},'gpx')">GPX</button>
        <button class="btn-sm btn-sm-green" onclick="event.stopPropagation();exportTrack(${t.id},'csv')">CSV</button>
        <button class="btn-sm btn-sm-pink" onclick="event.stopPropagation();deleteTrack(${t.id})">✕</button>
      </div>
    </div>`).join("");
}

function focusTrack(id) {
  selectedTrackId = id; renderTrackList();
  const t = tracks.find(t => t.id === id);
  if (!t || !t.points.length) return;
  map.fitBounds(L.latLngBounds(t.points.map(p => [p.lat, p.lon])), { padding: [40, 40] });
}

function deleteTrack(id) {
  if (!confirm("この記録を削除しますか？")) return;
  if (trackPolylines[id]) { map.removeLayer(trackPolylines[id]); delete trackPolylines[id]; }
  tracks = tracks.filter(t => t.id !== id);
  if (selectedTrackId === id) selectedTrackId = null;
  saveTracks(); renderTrackList(); toast("削除しました");
}

function exportTrack(id, fmt) {
  const t = tracks.find(t => t.id === id); if (!t) return;
  if (fmt === "gpx") downloadFile(toGPX(t), `${t.name}.gpx`, "application/gpx+xml");
  else downloadFile(toCSV(t), `${t.name}.csv`, "text/csv");
}

function toGPX(t) {
  const pts = t.points.map(p => `    <trkpt lat="${p.lat}" lon="${p.lon}"><ele>0</ele><time>${new Date(p.ts).toISOString()}</time></trkpt>`).join("\n");
  return `<?xml version="1.0" encoding="UTF-8"?>\n<gpx version="1.1" creator="PostingTracker">\n  <trk><n>${t.name}</n><trkseg>\n${pts}\n  </trkseg></trk>\n</gpx>`;
}

function toCSV(t) { return "latitude,longitude,timestamp\n" + t.points.map(p => `${p.lat},${p.lon},${new Date(p.ts).toISOString()}`).join("\n"); }

function downloadFile(content, name, mime) {
  const a = Object.assign(document.createElement("a"), { href: URL.createObjectURL(new Blob([content], { type: mime })), download: name });
  a.click(); URL.revokeObjectURL(a.href);
}

function loadGeoJSON(e) {
  const file = e.target.files[0]; if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => { try { addAreaLayer(JSON.parse(ev.target.result), file.name); toast(`読み込み完了: ${file.name}`); } catch { toast("GeoJSONの読み込みに失敗しました"); } };
  reader.readAsText(file); e.target.value = "";
}

function addAreaLayer(geojson, filename) {
  const color = AREA_COLORS[areaLayers.length % AREA_COLORS.length];
  const layer = L.geoJSON(geojson, {
    style: { color, weight: 2, opacity: 0.9, fillColor: color, fillOpacity: 0.08 },
    onEachFeature: (feature, layer) => { const name = feature.properties?.name || feature.properties?.エリア名 || ""; if (name) layer.bindTooltip(name, { permanent: false, direction: "center" }); }
  }).addTo(map);
  areaLayers.push({ id: Date.now(), name: filename.replace(/\.geojson?$/i, ""), color, featureCount: geojson.features?.length || 0, layer });
  map.fitBounds(layer.getBounds(), { padding: [20, 20] });
  renderAreaList();
}

function clearAreas() { areaLayers.forEach(a => map.removeLayer(a.layer)); areaLayers = []; renderAreaList(); toast("エリアを消去しました"); }

function renderAreaList() {
  const el = document.getElementById("area-list");
  if (areaLayers.length === 0) { el.innerHTML = `<div style="color:var(--muted);font-size:11px;text-align:center;padding:8px">エリアデータなし</div>`; return; }
  el.innerHTML = areaLayers.map(a => `
    <div class="area-item">
      <div class="area-color-bar" style="background:${a.color}"></div>
      <div style="flex:1;font-size:11px">${a.name} <span style="color:var(--muted)">(${a.featureCount}エリア)</span></div>
      <button class="btn-sm btn-sm-pink" onclick="removeArea(${a.id})">✕</button>
    </div>`).join("");
}

function removeArea(id) {
  const idx = areaLayers.findIndex(a => a.id === id); if (idx < 0) return;
  map.removeLayer(areaLayers[idx].layer); areaLayers.splice(idx, 1); renderAreaList();
}

function downloadSampleGeoJSON() {
  const sample = { type: "FeatureCollection", features: [
    { type: "Feature", properties: { name: "西京01A" }, geometry: { type: "Polygon", coordinates: [[[135.700,35.000],[135.710,35.000],[135.710,34.993],[135.700,34.993],[135.700,35.000]]] } },
    { type: "Feature", properties: { name: "西京01B" }, geometry: { type: "Polygon", coordinates: [[[135.710,35.000],[135.720,35.000],[135.720,34.993],[135.710,34.993],[135.710,35.000]]] } },
    { type: "Feature", properties: { name: "南区01A" }, geometry: { type: "Polygon", coordinates: [[[135.740,34.990],[135.755,34.990],[135.755,34.980],[135.740,34.980],[135.740,34.990]]] } }
  ]};
  downloadFile(JSON.stringify(sample, null, 2), "sample_areas.geojson", "application/json");
  toast("サンプルをダウンロードしました");
}

function goToCurrentLocation() {
  if (!navigator.geolocation) { toast("GPSが使えません"); return; }
  const btn = document.getElementById("locate-btn");
  btn.classList.add("locating");
  navigator.geolocation.getCurrentPosition(
    pos => { map.setView([pos.coords.latitude, pos.coords.longitude], 17, { animate: true }); btn.classList.remove("locating"); toast("現在地に移動しました"); },
    err => { btn.classList.remove("locating"); toast("現在地を取得できません"); },
    { enableHighAccuracy: true, timeout: 10000 }
  );
}

function calcDistance(points) {
  let total = 0;
  for (let i = 1; i < points.length; i++) {
    const R = 6371000, dLat = (points[i].lat - points[i-1].lat) * Math.PI / 180, dLon = (points[i].lon - points[i-1].lon) * Math.PI / 180;
    const a = Math.sin(dLat/2)**2 + Math.cos(points[i-1].lat*Math.PI/180) * Math.cos(points[i].lat*Math.PI/180) * Math.sin(dLon/2)**2;
    total += R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
  }
  return total;
}

function formatTime(s) { return [Math.floor(s/3600), Math.floor(s%3600/60), s%60].map(v => v.toString().padStart(2,"0")).join(":"); }
